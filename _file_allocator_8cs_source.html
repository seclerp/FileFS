150  private bool TryFindGap(int size, out Cursor cursor)
151  {
152  _logger.Information($"Trying to find existing gap of {size} bytes");
153 
154  cursor = default;
155 
156  // 1. Get all descriptors
157  var descriptors = _entryDescriptorRepository.ReadAll();
158 
159  // 2. Sort by offset
160  var orderedDescriptors = descriptors
161  .Where(descriptor => descriptor.Value.DataLength > 0)
162  .OrderBy(descriptor => descriptor.Value.DataOffset).ToArray();
163 
164  // 3. Check first gap
165  if (orderedDescriptors.Length > 0 && orderedDescriptors[0].Value.DataOffset >= size)
166  {
167  // Start of storage is our gap
168  cursor = new Cursor(0, SeekOrigin.Begin);
169 
170  _logger.Information($"Found gap of size {size} bytes at offset {cursor.Offset}");
171 
172  return true;
173  }
174 
175  // 4. Check other gaps
176  // We should find minimal gap but bigger or equals in size
177  var minimalGapSize = int.MaxValue;
178  var minimalGapCursor = default(Cursor);
179  var found = false;
180  for (var i = 0; i < orderedDescriptors.Length - 1; i++)
181  {
182  var current = orderedDescriptors[i];
183  var next = orderedDescriptors[i + 1];
184  var currentEnd = current.Value.DataOffset + current.Value.DataLength;
185  var nextStart = next.Value.DataOffset;
186 
187  // If we have gap with specific size or between files
188  var currentGapSize = nextStart - currentEnd;
189  if (currentGapSize >= size && currentGapSize < minimalGapSize)
190  {
191  minimalGapSize = currentGapSize;
192  minimalGapCursor = new Cursor(currentEnd, SeekOrigin.Begin);
193 
194  // We should know that we found at least one correct gap
195  found = true;
196  }
197  }
198

201  cursor = minimalGapCursor;
202 
203  _logger.Information($"Found gap of size {size} bytes at offset {cursor.Offset}");
204 
205  return true;
206  }
207
