46  public int Optimize()
47  {
48  return _storageOperationLocker.GlobalLock(() =>
49  {
50  _logger.Information("Start optimization process");
51 
52  var dataItemsMoved = 0;
53  var initialDataSize = _filesystemDescriptorAccessor.Value.FilesDataLength;
54 
55  // 1. Get all descriptors
56  var descriptors = _entryDescriptorRepository.ReadAll();
57 
58  _logger.Information($"There is {descriptors.Count} descriptors found");
59 
60  // 2. Sort by offset, filter zero-sized data because it is not necessary to move empty data
61  var orderedDescriptors = descriptors
62  .Where(descriptor => descriptor.Value.DataLength > 0)
63  .OrderBy(descriptor => descriptor.Value.DataOffset)
64  .ToArray();
65 
66  // 3a. Move first
67  if (orderedDescriptors.Length > 0 && orderedDescriptors[0].Value.DataOffset != 0)
68  {
69  ProcessGap(orderedDescriptors, 0, 0);
70  dataItemsMoved++;
71  }
72 
73  // 3b. Iterate over all descriptors to find gaps
74  for (var i = 0; i < orderedDescriptors.Length - 1; i++)
75  {
76  var current = orderedDescriptors[i];
77  var next = orderedDescriptors[i + 1];
78  var currentEnd = current.Value.DataOffset + current.Value.DataLength;
79  var nextStart = next.Value.DataOffset;
80 
81  // 4. Found a gap, write data of second right after first
82  if (nextStart - currentEnd > 0)
83  {
84  ProcessGap(orderedDescriptors, i + 1, currentEnd);
85  dataItemsMoved++;
86  }
87  }
88 
89  // 4. Update new size of file data in descriptor.
90  var allEntriesDataLength = orderedDescriptors.Select(descriptor => descriptor.Value.DataLength).Sum();
91 
92  _filesystemDescriptorAccessor.Update(filesDataLengthUpdater: _ => allEntriesDataLength);
93 
94  var bytesOptimized = initialDataSize - allEntriesDataLength;
95 
96  _logger.Information(
97  $"Optimization process completed, {dataItemsMoved} items moved, {bytesOptimized} bytes optimized");
98 
99  return bytesOptimized;
100  });
101  }
