251  private static void InitializeStorage(IServiceProvider serviceProvider, int storageSize, params IFileEntry[] itemsToCreate)
252  {
253  var fileRepository = serviceProvider.GetRequiredService<IFileRepository>();
254  var storageInitializer = serviceProvider.GetRequiredService<IStorageInitializer>();
255  storageInitializer.Initialize(storageSize, FileNameLength);
256
